  def initialize(nxt)
    @nxt          = nxt
    
    # defaults the same as NXT-G
    @ports        = [:b, :c]
    @direction    = :forward
    @steering     = {:left_motor => :b, :right_motor => :c, :direction => :straight, :power => 100}

    @duration     = {:rotations => 1}
    @next_action  = :brake
  end
