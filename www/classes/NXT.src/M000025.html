        def method_missing(method, *args, &block)
                name = method.id2name
                if /^motor_([abc])$/ =~ name
                        motor($1, block)
                elsif /^motors_([abc]+?)$/ =~ name
                        motors($1, block)
                else
                        raise "Unknown method '#{method}'"
                end
        end
