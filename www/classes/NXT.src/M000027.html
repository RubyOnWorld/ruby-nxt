        def motor(id, proc)
                id = id.intern if id.kind_of? String
                
                @motor_threads[id].join if (@motor_threads[id] and @motor_threads[id].alive?)
                
                t = Thread.new(@motors[id]) do |m|
                        proc.call(m)
                end
                
                @motor_threads[id] = t
        end
