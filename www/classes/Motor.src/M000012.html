        def update_state
                @log.debug(:update_state) { "getting state"}
                r = @nxt.GetOutputState(@port)
                @log.debug(:update_state) { "got state" }
        
                @state = {
                        :port => r[0],
                        :power => r[1],
                        :mode => r[2],
                        :regulation => r[3],
                        :ratio => r[4],
                        :run_state => r[5],
                        :degree_limit => r[6],
                        :tacho_count => r[7],
                        :tacho_count_block => r[8],
                        :degree_count => r[9]
                }
                
                debug(@state, :update_state)
        end
