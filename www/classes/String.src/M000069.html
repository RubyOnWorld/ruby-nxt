  def to_freq
    case self.downcase
      when "c"
        523
      when "c#"
        554
      when "d"
        587
      when "d#"
        622
      when "e"
        659
      when "f"
        698
      when "f#"
        740
      when "g"
        784
      when "g#"
        830
      when "a"
        880
      when "a#"
        923
      when "b"
        988
      else
        raise "Unknown Note: #{self}"
    end
  end
